Pipe Operator

2. lowercase

{{ message | lowercase }}

3. date:'fullDate'

{{ person.dob | date: "fullDate" }}

4. Pipe Chaining

{{ person.dob | date: "fullDate" | uppercase }}

5. json

{{ company | json }}

6. currency

{{ cur | currency: "USD":true:"2.2-4" }}

7. Percent

{{ num | percent: "1.2-5" }}

8. Decimal

{{ num | number: "3.2-5" }}
